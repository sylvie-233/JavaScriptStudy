### vite.config.js
```js
{
    build: { // 配置构建行为
        assetsDir: 静态资源目录
        outDir: 输出目录
        rollupOptions: rollup配置
    }
    css: { // 对CSS的行为进行配置
        module: 模块化配置
        preprocessorOptions: 预处理器配置
        devSourcemap: css的sourceMap文件索引
        postcss: PostCss配置相关
    }
    resolve: { // 文件导入别名
        "@": "/src"
    }
}
```
